#!/bin/bash

# Can only run from this script
export CALLED_FROM_START_APP=yes:wq

# Easier color
black='\e[30m'
red='\e[31m'
green='\e[32m'
yellow='\e[33m'
blue='\e[34m'
magenta='\e[35m'
cyan='\e[36m'
gray='\e[37m'
white='\e[0m'

clear

echoc()
{
  w=$(stty size | cut -d" " -f2)       # width of the terminal
  l=${#1}                              # length of the string
  printf "%"$((l+(w-l)/2))"s\n" "$1"   # print string padded to proper width (%Ws)
}

while : #Start loop
do

	clear

	echo -e $cyan'Master menu'$white
	option=( 'File menu(1)' 'User menu(2)' 'Location menu(3)' 'Fun menu(4)' 'Process menu(5)' 'Exit(6)' 'Shutdown(7)' )
	# List options for the user
	echo '===================================================='
	for element in "${option[@]}"
	do
		echo -e $green": $element"$white
	done
	echo '===================================================='

	read -p 'What to you want to do: ' act
	master=${act,,}

	echo ' '
	
	# Case for menu options
	case $master in
		file | 1)
			echo $green'File menu'$white #opens file menu
			echo ' '
			sleep 0.01
			bash menu/filemenu.txt
			;;
		user | 2)
			echo $green'User menu'$white #opens user meu
			echo ' '
			sleep 0.01
			bash menu/usermenu.txt
			;;
		find | 3)
			echo $green'Locating menu'$white #opens find menu
			echo ' '
			sleep 0.01
			bash menu/findmenu.txt
			;;
		fun | 4)
			echo $green'Fun menu'$white #opens fun menu
			echo ' '
			sleep 0.01
			bash menu/funmenu.txt
			;;
		process | 5)
			echo $green'Process menu'$white #opens process menu
			echo ' '
			sleep 0.01
			bash menu/processmenu.txt
			;;
		exit | 6)
			exit #exits from program
			;;
		shutdown | shut | 7)
			shutdown now #shuts down
			;;
		*)
			echo -e '\e[31mWrong input\e[0m' #iinforms the user if the user exists
			echo ' '
			read -p 'Continue(Press Enter)'
			;;
	esac
done
