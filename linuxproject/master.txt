#!/bin/bash

# Can only run from this script
export CALLED_FROM_START_APP=yes:wq

# Easier color
black='\e[30m'
red='\e[31m'
green='\e[32m'
yellow='\e[33m'
blue='\e[34m'
magenta='\e[35m'
cyan='\e[36m'
gray='\e[37m'
white='\e[0m'

clear

while : #Start loop
do

	clear

	echo -e $cyan'Master menu'$white
	option=( 'File menu(1)' 'User menu(2)' 'Location menu(3)' 'Fun menu(4)' 'Process menu(5)' 'Exit(6)' 'Shutdown(7)' )
	# List options for the user
	echo ====================================================
	for element in "${option[@]}"
	do
		echo -e $green": $element"$white
	done
	echo ====================================================

	read -p 'What to you want to do: ' act
	master=${act,,}

	clear
	
	# Case for menu options
	case $master in
		file | 1)
			echo 'File menu'
			echo ' '
			sleep 0.01
			bash menu/filemenu.txt
			;;
		user | 2)
			echo 'User menu'
			echo ' '
			sleep 0.01
			bash menu/usermenu.txt
			;;
		find | 3)
			echo 'Locating menu'
			;;
		fun | 4)
			echo 'Fun menu'
			;;
		process | 5)
			echo 'Process menu'
			;;
		exit | 6)
			exit
			;;
		shutdown | shut | 7)
			shutdown now
			;;
		*)
			echo -e '\e[31mWrong input\e[0m'
			sleep 3
			;;
	esac
done
